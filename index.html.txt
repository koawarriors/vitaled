        // Authentication
        document.getElementById('loginForm').addEventListener('submit', function(e) {
            e.preventDefault();
            const email = document.getElementById('email').value;
            const password = document.getElementById('password').value;
            const institution = document.getElementById('institution').value;
            
            if (email && password && institution) {
                login('student', email);
            }
        });

        function login(role, email) {
            const loginScreen = document.getElementById('loginScreen');
            const studentDashboard = document.getElementById('studentDashboard');
            const adminDashboard = document.getElementById('adminDashboard');
            
            loginScreen.style.display = 'none';
            
            if (role === 'admin') {
                adminDashboard.style.display = 'block';
                studentDashboard.style.display = 'none';
            } else {
                studentDashboard.style.display = 'block';
                adminDashboard.style.display = 'none';
                
                // Set user info
                const name = email.split('@')[0].replace('.', ' ');
                const formattedName = name.charAt(0).toUpperCase() + name.slice(1);
                document.getElementById('welcomeName').textContent = formattedName.split(' ')[0];
                document.getElementById('studentName').textContent = formattedName;
                document.getElementById('studentAvatar').textContent = formattedName.split(' ').map(n => n[0]).join('').toUpperCase();
            }
        }

        function loginAsAdmin() {
            login('admin', 'admin@vitaled.com');
        }

        function logout() {
            document.getElementById('loginScreen').style.display = 'flex';
            document.getElementById('studentDashboard').style.display = 'none';
            document.getElementById('adminDashboard').style.display = 'none';
            document.getElementById('loginForm').reset();
        }

        }
